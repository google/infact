00047 void
00048 Interpreter::Eval(StreamTokenizer &st) {
00049   // Keeps reading assignment statements until there are no more tokens.
00050   while (st.PeekTokenType() != StreamTokenizer::EOF_TYPE) {
00051 #ifdef INFACT_THROW_EXCEPTIONS
00052     try {
00053 #endif
00054     // Read variable name or type specifier.
00055     StreamTokenizer::TokenType token_type = st.PeekTokenType();
00056     VarMapBase *varmap = env_->GetVarMapForType(st.Peek());
00057     bool is_type_specifier =  varmap != nullptr;
00058     if (token_type != StreamTokenizer::IDENTIFIER && !is_type_specifier) {
00059       string expected_type =
00060           string(StreamTokenizer::TypeName(StreamTokenizer::IDENTIFIER)) +
00061           " or type specifier";
00062       string found_type = StreamTokenizer::TypeName(token_type);
00063       WrongTokenTypeError(st.PeekTokenStart(), expected_type, found_type,
00064                           st.Peek());
00065     }
00066 
00067     string type = "";
00068     if (is_type_specifier) {
00069       // Consume and remember the type specifier.
00070       st.Next();              // Explicit type could be a concrete type.
00071       type = varmap->Name();  // Remember the abstract type.
00072 
00073       // Check that next token is a variable name.
00074       StreamTokenizer::TokenType token_type = st.PeekTokenType();
00075       if (token_type != StreamTokenizer::IDENTIFIER) {
00076         WrongTokenTypeError(st.PeekTokenStart(), StreamTokenizer::IDENTIFIER,
00077                             token_type, st.Peek());
00078       }
00079     }
00080 
00081     string varname = st.Next();
00082 
00083     // Next, read equals sign.
00084     token_type = st.PeekTokenType();
00085     if (st.Peek() != "=") {
00086       WrongTokenError(st.PeekTokenStart(), "=", st.Peek(), st.PeekTokenType());
00087     }
00088 
00089     // Consume equals sign.
00090     st.Next();
00091 
00092     if (st.PeekTokenType() == StreamTokenizer::EOF_TYPE) {
00093       ostringstream err_ss;
00094       err_ss << "Interpreter:" << filename_
00095              << ": error: unexpected EOF at stream position "
00096              << st.tellg();
00097       Error(err_ss.str());
00098     }
00099 
00100     // Consume and set the value for this variable in the environment.
00101     env_->ReadAndSet(varname, st, type);
00102 
00103     token_type = st.PeekTokenType();
00104     if (st.Peek() != ";") {
00105       WrongTokenError(st.PeekTokenStart(), ";", st.Peek(), st.PeekTokenType());
00106     }
00107     // Consume semicolon.
00108     st.Next();
00109 #ifdef INFACT_THROW_EXCEPTIONS
00110     }
00111     catch (std::runtime_error &e) {
00112       cerr << "threw exception: " << e.what() << endl;
00113       // For now, we simply give up.
00114       break;
00115     }
00116 #endif
00117   }
00118 }
