00041 EnvironmentImpl::EnvironmentImpl(int debug) {
00042   debug_ = debug;
00043 
00044   // Set up VarMap instances for each of the primitive types and their vectors.
00045   var_map_["bool"] = new VarMap<bool>("bool", this);
00046   var_map_["int"] = new VarMap<int>("int", this);
00047   var_map_["double"] = new VarMap<double>("double", this);
00048   var_map_["string"] = new VarMap<string>("string", this);
00049   var_map_["bool[]"] = new VarMap<vector<bool> >("bool[]", "bool", this);
00050   var_map_["int[]"] = new VarMap<vector<int> >("int[]", "int", this);
00051   var_map_["double[]"] =
00052       new VarMap<vector<double> >("double[]", "double", this);
00053   var_map_["string[]"] =
00054       new VarMap<vector<string> >("string[]", "string", this);
00055 
00056   // Set up VarMap instances for each of the Factory-constructible types
00057   // and their vectors.
00058   for (FactoryContainer::iterator factory_it = FactoryContainer::begin();
00059        factory_it != FactoryContainer::end(); ++factory_it) {
00060     unordered_set<string> registered;
00061     (*factory_it)->CollectRegistered(registered);
00062     string base_name = (*factory_it)->BaseName();
00063 
00064     // Create type-specific VarMap from the Factory and add to var_map_.
00065     VarMapBase *obj_var_map = (*factory_it)->CreateVarMap(this);
00066     var_map_[obj_var_map->Name()] = obj_var_map;
00067 
00068     if (debug_ >= 2) {
00069       cerr << "Environment: created VarMap for " << obj_var_map->Name()
00070            << endl;
00071     }
00072 
00073     // Create VarMap for vectors of shared_object of T and add to var_map_.
00074     VarMapBase *obj_vector_var_map = (*factory_it)->CreateVectorVarMap(this);
00075     var_map_[obj_vector_var_map->Name()] = obj_vector_var_map;
00076 
00077     if (debug_ >= 2) {
00078       cerr << "Environment: created VarMap for " << obj_vector_var_map->Name()
00079            << endl;
00080     }
00081 
00082     for (unordered_set<string>::const_iterator it = registered.begin();
00083          it != registered.end(); ++it) {
00084       const string &concrete_type_name = *it;
00085 
00086       unordered_map<string, string>::const_iterator concrete_to_factory_it =
00087           concrete_to_factory_type_.find(concrete_type_name);
00088       if (concrete_to_factory_it != concrete_to_factory_type_.end()) {
00089         // Warn user that there are two entries for the same concrete type

00091         cerr << "Environment: WARNING: trying to override existing "
00092              << "concrete-to-factory type mapping ["
00093              << concrete_type_name << " --> " << concrete_to_factory_it->second

00109 EnvironmentImpl::ReadAndSet(const string &varname, StreamTokenizer &st,
00110                             const string type) {
00111   bool is_vector =
00112       st.PeekTokenType() == StreamTokenizer::RESERVED_CHAR &&
00113       st.Peek() == "{";
00114 
00115   if (is_vector) {
00116     // Consume open brace.
00117     st.Next();
00118   } else if (st.PeekTokenType() == StreamTokenizer::RESERVED_CHAR ||
00119              (st.PeekTokenType() == StreamTokenizer::RESERVED_WORD &&
00120               st.Peek() != "true" && st.Peek() != "false" &&
00121               st.Peek() != "nullptr" && st.Peek() != "NULL")) {
00122     ostringstream err_ss;
00123     err_ss << "Environment: error: expected literal or Factory-constructible "
00124            << "type but found token \"" << st.Peek() << "\" of type "
00125            << StreamTokenizer::TypeName(st.PeekTokenType());
00126     Error(err_ss.str());
00127   }
00128 
00129   string next_tok = st.Peek();
00130   bool is_object_type = false;
00131 
00132   string inferred_type = InferType(varname, st, is_vector, &is_object_type);
00133 
00134   if (is_vector) {
00135     st.Putback();
00136     next_tok = st.Peek();
00137   }
00138 
00139   if (debug_ >= 1) {
00140     cerr << "Environment::ReadAndSet: next_tok=\"" << next_tok
00141          << "\"; explicit type=\"" << type << "\"; "
00142          << "inferred_type=\"" << inferred_type << "\"" << endl;
00143   }
00144 
00145   if (type == "" && inferred_type == "") {
00146     ostringstream err_ss;
00147     err_ss << "Environment: error: no explicit type specifier and could not "
00148            << "infer type for variable " << varname;
00149     Error(err_ss.str());
00150   }
00151   if (type != "" && inferred_type != "" && type != inferred_type) {
00152     ostringstream err_ss;
00153     err_ss << "Environment: error: explicit type " << type
00154            << " and inferred type " << inferred_type
00155            << " disagree for variable " << varname;
00156     Error(err_ss.str());
00157   }
00158 
00159   // If no explicit type specifier, then the inferred_type is the type.
00160   string varmap_type = type == "" ? inferred_type : type;
00161 
00162   // Check that varmap_type is a key in var_map_.
00163   var_map_[varmap_type]->ReadAndSet(varname, st);
00164   types_[varname] = varmap_type;
00165 }

00167 string
00168 EnvironmentImpl::InferType(const string &varname,
00169                            const StreamTokenizer &st, bool is_vector,
00170                            bool *is_object_type) {
00171   *is_object_type = false;
00172   string next_tok = st.Peek();
00173   switch (st.PeekTokenType()) {
00174     case StreamTokenizer::RESERVED_WORD:
00175       if (next_tok == "true" || next_tok == "false") {
00176         return is_vector ? "bool[]" : "bool";
00177       } else {
00178         return "";
00179       }
00180       break;
00181     case StreamTokenizer::STRING:
00182       return is_vector ? "string[]" : "string";
00183       break;
00184     case StreamTokenizer::NUMBER:
00185       {
00186         // If a token is a NUMBER, it is a double iff it contains a
00187         // decimal point.
00188         size_t dot_pos = next_tok.find('.');
00189         if (dot_pos != string::npos) {
00190           return is_vector ? "double[]" : "double";
00191         } else {
00192           return is_vector ? "int[]" : "int";
00193         }
00194       }
00195       break;
00196     case StreamTokenizer::IDENTIFIER:
00197       {
00198         string type = "";
00199 
00200         // Find out if next_tok is a concrete typename or a variable.
00201         unordered_map<string, string>::const_iterator factory_type_it =
00202             concrete_to_factory_type_.find(next_tok);
00203         unordered_map<string, string>::const_iterator var_type_it =
00204             types_.find(next_tok);
00205         if (factory_type_it != concrete_to_factory_type_.end()) {
00206           // Set type to be abstract factory type.
00207           if (debug_ >= 1) {
00208             cerr << "Environment::InferType: concrete type is " << next_tok
00209                  << "; mapping to abstract Factory type "
00210                  << factory_type_it->second << endl;
00211           }
00212           type = factory_type_it->second;
00213           *is_object_type = true;
00214           type = is_vector ? type + "[]" : type;
00215 
00216           if (debug_ >= 1) {
00217             cerr << "Environment::InferType: type "
00218                  << (is_vector ? "is" : "isn't")
00219                  << " a vector, so final inferred type is " << type << endl;
00220           }
00221         } else if (var_type_it != types_.end()) {
00222           // Could be a variable, in which case we need not only to return
00223           // the variable's type, but also set is_object_type and is_vector
00224           // based on the variable's type string.
00225           string append = is_vector ? "[]" : "";
00226           type = var_type_it->second + append;
00227           if (debug_ >= 1) {
00228             cerr << "Environment::InferType: found variable "
00229                  << var_type_it->first << " of type " << var_type_it->second
00230                  << "; type is " << type << endl;
00231           }
00232         } else {
00233           ostringstream err_ss;
00234           err_ss << "Environment: error: token " << next_tok
00235                  << " is neither a variable nor a concrete object typename";
00236           Error(err_ss.str());
00237         }
00238         return type;
00239       }
00240       break;
00241     default:
00242       return "";
00243   }
00244   return "";
00245 }
