762  shared_ptr<T> CreateOrDie(StreamTokenizer &st, Environment *env = nullptr) {
763  shared_ptr<Environment> env_ptr(env == nullptr ?
764  Environment::CreateEmpty() : env->Copy());
765  size_t start = st.PeekTokenStart();
766  StreamTokenizer::TokenType token_type = st.PeekTokenType();
767  if (token_type == StreamTokenizer::RESERVED_WORD &&
768  (st.Peek() == "nullptr" || st.Peek() == "NULL")) {
769  // Consume the nullptr.
770  st.Next();
771  return shared_ptr<T>();
772  }
773  if (token_type != StreamTokenizer::IDENTIFIER) {
774  ostringstream err_ss;
775  err_ss << "Factory<" << BaseName() << ">: "
776  << "error: expected type specifier token but found "
777  << StreamTokenizer::TypeName(token_type);
778  Error(err_ss.str());
779  }
780 
781  // Read the concrete type of object to be created.
782  string type = st.Next();
783 
784  // Read the open parenthesis token.
785  if (st.Peek() != "(") {
786  ostringstream err_ss;
787  err_ss << "Factory<" << BaseName() << ">: "
788  << "error: expected '(' at stream position "
789  << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
790  Error(err_ss.str());
791  }
792  st.Next();
793 
794  // Attempt to create an instance of type.
795  typename unordered_map<string, const Constructor<T> *>::iterator cons_it =
796  cons_table_->find(type);
797  if (cons_it == cons_table_->end()) {
798  ostringstream err_ss;
799  err_ss << "Factory<" << BaseName() << ">: "
800  << "error: unknown type: \"" << type << "\"";
801  Error(err_ss.str());
802  }
803  shared_ptr<T> instance(cons_it->second->NewInstance());
804 
805  // Ask new instance to set up member initializers.
806  Initializers initializers;
807  instance->RegisterInitializers(initializers);
808 
809  // Parse initializer list.
810  while (st.Peek() != ")") {
811  token_type = st.PeekTokenType();
812  if (token_type != StreamTokenizer::IDENTIFIER) {
813  ostringstream err_ss;
814  err_ss << "Factory<" << BaseName() << ">: "
815  << "error: expected token of type IDENTIFIER at "
816  << "stream position " << st.PeekTokenStart() << " but found "
817  << StreamTokenizer::TypeName(token_type) << ": \""
818  << st.Peek() << "\"";
819  Error(err_ss.str());
820  }
821  size_t member_name_start = st.PeekTokenStart();
822  string member_name = st.Next();
823  typename Initializers::iterator init_it = initializers.find(member_name);
824  if (init_it == initializers.end()) {
825  ostringstream err_ss;
826  err_ss << "Factory<" << BaseName() << ">: "
827  << "error: unknown member name \"" << member_name
828  << "\" in initializer list for type " << type << " at stream "
829  << "position " << member_name_start;
830  Error(err_ss.str());
831  }
832  MemberInitializer *member_initializer = init_it->second;
833 
834  // Read open parenthesis.
835  if (st.Peek() != "(") {
836  ostringstream err_ss;
837  err_ss << "Factory<" << BaseName() << ">: "
838  << "error: expected '(' at stream position "
839  << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
840  Error(err_ss.str());
841  }
842  st.Next();
843 
844  // Initialize member based on following token(s).
845  member_initializer->Init(st, env_ptr.get());
846 
847  // Read close parenthesis for current member initializer.
848  if (st.Peek() != ")") {
849  ostringstream err_ss;
850  err_ss << "Factory<" << BaseName() << ">: "
851  << "error: expected ')' at stream position "
852  << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
853  Error(err_ss.str());
854  }
855  st.Next();
856 
857  // Each member initializer must be followed by a comma or the final
858  // closing parenthesis.
859  if (st.Peek() != "," && st.Peek() != ")") {
860  ostringstream err_ss;
861  err_ss << "Factory<" << BaseName() << ">: "
862  << "error: expected ',' or ')' at stream position "
863  << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
864  Error(err_ss.str());
865  }
866  // Read comma, if present.
867  if (st.Peek() == ",") {
868  st.Next();
869  }
870  }
871 
872  // Read the close parenthesis token for this factory type specification.
873  if (st.Peek() != ")") {
874  ostringstream err_ss;
875  err_ss << "Factory<" << BaseName() << ">: "
876  << "error: expected ')' at stream position "
877  << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
878  Error(err_ss.str());
879  }
880  st.Next();
881 
882  // Run through all member initializers: if any are required but haven't
883  // been invoked, it is an error.
884  for (typename Initializers::const_iterator init_it = initializers.begin();
885  init_it != initializers.end();
886  ++init_it) {
887  MemberInitializer *member_initializer = init_it->second;
888  if (member_initializer->Required() &&
889  member_initializer->Initialized() == 0) {
890  ostringstream err_ss;
891  err_ss << "Factory<" << BaseName() << ">: "
892  << "error: initialization for member with name \""
893  << init_it->first << "\" required but not found (current "
894  << "stream position: " << st.tellg() << ")";
895  Error(err_ss.str());
896  }
897  }
898 
899  size_t end = st.tellg();
900  // Invoke new instance's Init method.
901  string stream_str = st.str();
902  //cerr << "Full stream string is: \"" << stream_str << "\"" << endl;
903  string init_str = stream_str.substr(start, end - start);
904  //cerr << "PostInit string is: \"" << init_str << "\"" << endl;
905  instance->PostInit(env_ptr.get(), init_str);
906 
907  return instance;
908  }
