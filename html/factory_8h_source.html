676  shared_ptr<T> CreateOrDie(StreamTokenizer &st, Environment *env = nullptr) {
677  shared_ptr<Environment> env_ptr(env == nullptr ?
678  Environment::CreateEmpty() : env->Copy());
679  size_t start = st.PeekTokenStart();
680  StreamTokenizer::TokenType token_type = st.PeekTokenType();
681  if (token_type == StreamTokenizer::RESERVED_WORD &&
682  (st.Peek() == "nullptr" || st.Peek() == "NULL")) {
683  // Consume the nullptr.
684  st.Next();
685  return shared_ptr<T>();
686  }
687  if (token_type != StreamTokenizer::IDENTIFIER) {
688  ostringstream err_ss;
689  err_ss << "Factory<" << BaseName() << ">: "
690  << "error: expected type specifier token but found "
691  << StreamTokenizer::TypeName(token_type);
692  Error(err_ss.str());
693  }
694 
695  // Read the concrete type of object to be created.
696  string type = st.Next();
697 
698  // Read the open parenthesis token.
699  if (st.Peek() != "(") {
700  ostringstream err_ss;
701  err_ss << "Factory<" << BaseName() << ">: "
702  << "error: expected '(' at stream position "
703  << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
704  Error(err_ss.str());
705  }
706  st.Next();
707 
708  // Attempt to create an instance of type.
709  typename unordered_map<string, const Constructor<T> *>::iterator cons_it =
710  cons_table_->find(type);
711  if (cons_it == cons_table_->end()) {
712  ostringstream err_ss;
713  err_ss << "Factory<" << BaseName() << ">: "
714  << "error: unknown type: \"" << type << "\"";
715  Error(err_ss.str());
716  }
717  shared_ptr<T> instance(cons_it->second->NewInstance());
718 
719  // Ask new instance to set up member initializers.
720  Initializers initializers;
721  instance->RegisterInitializers(initializers);
722 
723  // Parse initializer list.
724  while (st.Peek() != ")") {
725  token_type = st.PeekTokenType();
726  if (token_type != StreamTokenizer::IDENTIFIER) {
727  ostringstream err_ss;
728  err_ss << "Factory<" << BaseName() << ">: "
729  << "error: expected token of type IDENTIFIER at "
730  << "stream position " << st.PeekTokenStart() << " but found "
731  << StreamTokenizer::TypeName(token_type) << ": \""
732  << st.Peek() << "\"";
733  Error(err_ss.str());
734  }
735  size_t member_name_start = st.PeekTokenStart();
736  string member_name = st.Next();
737  typename Initializers::iterator init_it = initializers.find(member_name);
738  if (init_it == initializers.end()) {
739  ostringstream err_ss;
740  err_ss << "Factory<" << BaseName() << ">: "
741  << "error: unknown member name \"" << member_name
742  << "\" in initializer list for type " << type << " at stream "
743  << "position " << member_name_start;
744  Error(err_ss.str());
745  }
746  MemberInitializer *member_initializer = init_it->second;
747 
748  // Read open parenthesis.
749  if (st.Peek() != "(") {
750  ostringstream err_ss;
751  err_ss << "Factory<" << BaseName() << ">: "
752  << "error: expected '(' at stream position "
753  << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
754  Error(err_ss.str());
755  }
756  st.Next();
757 
758  // Initialize member based on following token(s).
759  member_initializer->Init(st, env_ptr.get());
760 
761  // Read close parenthesis for current member initializer.
762  if (st.Peek() != ")") {
763  ostringstream err_ss;
764  err_ss << "Factory<" << BaseName() << ">: "
765  << "error: expected ')' at stream position "
766  << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
767  Error(err_ss.str());
768  }
769  st.Next();
770 
771  // Each member initializer must be followed by a comma or the final
772  // closing parenthesis.
773  if (st.Peek() != "," && st.Peek() != ")") {
774  ostringstream err_ss;
775  err_ss << "Factory<" << BaseName() << ">: "
776  << "error: expected ',' or ')' at stream position "
777  << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
778  Error(err_ss.str());
779  }
780  // Read comma, if present.
781  if (st.Peek() == ",") {
782  st.Next();
783  }
784  }
785 
786  // Read the close parenthesis token for this factory type specification.
787  if (st.Peek() != ")") {
788  ostringstream err_ss;
789  err_ss << "Factory<" << BaseName() << ">: "
790  << "error: expected ')' at stream position "
791  << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
792  Error(err_ss.str());
793  }
794  st.Next();
795 
796  // Run through all member initializers: if any are required but haven't
797  // been invoked, it is an error.
798  for (typename Initializers::const_iterator init_it = initializers.begin();
799  init_it != initializers.end();
800  ++init_it) {
801  MemberInitializer *member_initializer = init_it->second;
802  if (member_initializer->Required() &&
803  member_initializer->Initialized() == 0) {
804  ostringstream err_ss;
805  err_ss << "Factory<" << BaseName() << ">: "
806  << "error: initialization for member with name \""
807  << init_it->first << "\" required but not found (current "
808  << "stream position: " << st.tellg() << ")";
809  Error(err_ss.str());
810  }
811  }
812 
813  size_t end = st.tellg();
814  // Invoke new instance's Init method.
815  string stream_str = st.str();
816  //cerr << "Full stream string is: \"" << stream_str << "\"" << endl;
817  string init_str = stream_str.substr(start, end - start);
818  //cerr << "PostInit string is: \"" << init_str << "\"" << endl;
819  instance->PostInit(env_ptr.get(), init_str);
820 
821  return instance;
822  }
